Table users {
  id int [pk, increment]
  username string [unique, not null]
  password string [not null]
  email email [not null]
}

Table profiles {
  user_id int [pk, fk]
  first_name string
  last_name string
  location place
  gender choice
  dob date
  reason text
  goal_cal int [not null]
}

Table groups {
  id int [pk, increment]
  name string [unique, not null]
  description text
}

Table follows {
  user_following_id int [pk, fk]
  user_followed_id int [pk, fk]
}

Table users_groups {
  group_id int [pk, fk]
  user_id int [fk]
}

Table comments {
  id int [pk, increment]
  user_id int [fk]
  text string [not null]
  timestamp datetime [not null]
}
